Boost 之日期和时间

总览

Boost中将时间（广义的）分为日期(1999-9-11)和时间(20:00:00)两个处理维度，关于日期boost提供格力高日历表示方法，在后面的命名空间中有所体现。关于时间现在标准库已经提供了std::chrono，boost里除了chronoa还有精度差一些的时间date_time。

Boost为两者都提供了，点、段、区间的概念、使用很方便且跨平台

计时器

先来看几个好用的计时器

#include <boost/timer.hpp>
void test_timer()
{
    // 《计时器》 时间记录的对象，用来记录时间流逝，对象初始化时开始计时，精度看起来只到微妙，也有最大时间限制
    boost::timer t;
    std::cout << "max timespan:" << t.elapsed_max() / 3600 << "h" << std::endl;
    
    std::cout << "min timespan:" << t.elapsed_min() << "s" << std::endl;

    Sleep(3000);	//Some 耗时操作

    std::cout << "now time elapsed:" << t.elapsed() << "s" << std::endl;
}

RAII方式的计时器

#include <boost/progress.hpp>
#include <sstream>
void test_progress_timer()
{
    std::stringstream ss;
    {
        boost::progress_timer pt(ss);	//继承自timer 构造参数默认是std::cout
        Sleep(1000);
        std::cout << "something done！" << std::endl;
    }
    std::cout << "done over! time:" << ss.str() << std::endl;
}

一个有点意思的进度显示器，这个作用还是有的，可以在日志里记录文件下载或写入进度，帮助排查问题

void test_progress_display()
{
    //有点作用，如:通过如日志记录文件下载的进度
    std::stringstream ss;
    static const int MAX_BASE_TIME = 100;
    boost::progress_display pd(MAX_BASE_TIME, ss, "总进度  :" , "进度条  :", "当前进度:");
    for (auto i = 0; i < MAX_BASE_TIME; ++i)
    {
        Sleep(100);
        ++pd;
        std::cout << ss.str() << std::endl;
    }
}

日期

话不多说，上代码

//关于日期 记录日期有极限(1400-01-01~9999-12-31)
#include <boost/date_time/gregorian/gregorian.hpp>		//处理日期
void test_date_time()
{
    //以gregorian(格力高日历)为基础，命名空间的由来,获取今天的日期
    std::cout << boost::gregorian::day_clock::local_day() << std::endl;
    std::cout << boost::gregorian::day_clock::universal_day() << std::endl;
    
    /* 都是一些日期的对象，可以参与运算
    * boost::gregorian::date d(2022,4,18)
    * boost::gregorian::years y(10);
    * boost::gregorian::months y(10);
    * boost::gregorian::weeks y(10);
    * boost::gregorian::days y(10);
    */

    typedef boost::gregorian::date Date;	//值得一提的是，它重载了> < =运算符，支持比较运算
    std::cout << Date(2024, 1, 13) << std::endl;
    std::cout << Date(2024, 1, 13).year() << std::endl;
    std::cout << Date(2024, 1, 13).month() << std::endl;
    std::cout << Date(2024, 1, 13).day() << std::endl;
    std::cout << Date(2024, 1, 13).day_count() << std::endl;//过了多少年
    std::cout << Date(2024, 1, 13).day_number() << std::endl;//过了多少年
    std::cout << Date(2024, 1, 13).day_of_week() << std::endl;//星期几
    std::cout << Date(2024, 1, 13).day_of_year() << std::endl;//过了多少天了
    std::cout << Date(2024, 1, 13).week_number() << std::endl;//这个月过了几周

    //可以与std::tm进行互转
    Date d(2024, 2, 13);
    std::tm tm = boost::gregorian::to_tm(d);
    Date d2 = boost::gregorian::date_from_tm(tm);
    std::cout << d2 << std::endl;
    /* std::tm 的结构
    * std::tm
    struct tm
    {
        int tm_sec;   // seconds after the minute - [0, 60] including leap second
        int tm_min;   // minutes after the hour - [0, 59]
        int tm_hour;  // hours since midnight - [0, 23]
        int tm_mday;  // day of the month - [1, 31]
        int tm_mon;   // months since January - [0, 11]
        int tm_year;  // years since 1900
        int tm_wday;  // days since Sunday - [0, 6]
        int tm_yday;  // days since January 1 - [0, 365]
        int tm_isdst; // daylight savings time flag
    };
    */


    //日期段 date_period
    boost::gregorian::date_period dp(Date(2022, 4, 18), Date(2024, 1, 13));
    std::cout << dp << std::endl;
    boost::gregorian::date_period dp1(Date(2022, 4, 28), boost::gregorian::date_duration(365*2));
    std::cout << dp1 << std::endl;

    //两端延长
    dp.expand(boost::gregorian::date_duration(365));
    std::cout << dp << std::endl;
    //平移
    dp.shift(boost::gregorian::date_duration(365));
    std::cout << dp << std::endl;
    /*
    * date_duration的一些成员函数， 返回bool值(除intersection)
    * is_before()/is_after() 区间是否在日期前、后
    * contains() 是否包含
    * intersects() 是否存在交集
    * intersection() 返回交集，无交集返回无效区间
    * is_adjacent() 日期区间是否相邻
    */

    //日期“迭代器”，很哇塞
    std::cout << "[日期迭代器的使用]:" << std::endl;
    //这里的迭代器和真正的迭代器仍然有差距，缺少differ_type,pointer,reference
    Date d3(2022, 4, 18);
    boost::gregorian::day_iterator di(d3);	
    boost::gregorian::week_iterator wi(d3);
    boost::gregorian::month_iterator mi(d3);
    boost::gregorian::year_iterator yi(d3);

    static const int MAX_ITERATER_COUNT = 3;
    std::cout << "day iterater:" << std::endl;
    for (size_t i = 0; i < MAX_ITERATER_COUNT; i++)
    {
        ++di; //只支持前置运算
        std::cout << di->year() << "-" << di->month() << "-" << di->day() << std::endl;
    }

    std::cout << "week iterater:" << std::endl;
    for (size_t i = 0; i < MAX_ITERATER_COUNT; i++)
    {
        ++wi; //只支持前置运算
        std::cout << wi->year() << "-" << wi->month() << "-" << wi->day() << std::endl;
    }

    std::cout << "month iterater:" << std::endl;
    for (size_t i = 0; i < MAX_ITERATER_COUNT; i++)
    {
        ++mi; //只支持前置运算
        std::cout << mi->year() << "-" << mi->month() << "-" << mi->day() << std::endl;
    }

    std::cout << "year iterater:" << std::endl;
    for (size_t i = 0; i < MAX_ITERATER_COUNT; i++)
    {
        ++yi; //只支持前置运算
        std::cout << yi->year() << "-" << yi->month() << "-" << yi->day() << std::endl;
    }

    //闰年的一些计算
    std::cout << "year 2024 is " << (boost::gregorian::gregorian_calendar::is_leap_year(2024) ? "" : "not ")
        << "a leap year" << std::endl;

    std::cout << "year 2024 eng of Feb has [" 
        << boost::gregorian::gregorian_calendar::end_of_month_day(2024, 2) << "] days" << std::endl;

}

关于时间

和之后的chrono在理念上相同

先看时间段的概念，可以参与时间运算

//关于时间，std::chrono似乎有了更好的方案
#include <boost/date_time/posix_time/posix_time.hpp>	//处理时间
#define BOOST_DATE_TIME_POSIX_TIME_STD_CONFIG //boost上支持纳秒的宏
//时间段
void test_time_duration()
{
    
    boost::posix_time::time_duration td(1,10,30,1000);
    std::cout << td << std::endl;
    //下面都是子类，time_duration可以进行运算,构造函数可以传入负数
    boost::posix_time::hours hour(1);
    boost::posix_time::minutes minu(1);
    boost::posix_time::seconds sec(1);
    boost::posix_time::milliseconds millisec(1); //some as boost::posix_time::millisec
    boost::posix_time::microseconds mircsec(1); //same as boost::posix_time::microsec
    boost::posix_time::nanoseconds nanosec(1); //same as boost::posix_time::nanosec
    std::cout << td + hour + minu + sec + millisec + mircsec + nanosec << std::endl;

}

时间点的概念

//时间点
void test_ptime()
{
    //ptime 类似与date，可以进行高效的拷贝赋值，支持全序比较和赋值
    //ptime相当于 ptime+time_duration
    std::cout << boost::posix_time::second_clock::local_time() << std::endl; //调用universal_time()一般情况下也一样
    std::cout << boost::posix_time::microsec_clock::local_time() << std::endl;

    typedef boost::gregorian::date Date;
    typedef boost::posix_time::ptime PTime;
    

    PTime pt(Date(2022,4,18), boost::posix_time::time_duration(1, 10, 30, 1000));
    std::cout << pt + boost::gregorian::days(10) + boost::gregorian::years(1) + boost::posix_time::hours(10) << std::endl;
    std::cout << pt.date() << std::endl;	//年月日
    std::cout << boost::posix_time::from_time_t(std::time(0)).date() << std::endl;// 支持从time_t单向转换
}

一个日期使用的小案例，可实现根据时间告诉你该做什么了，想象一下，可以方便实现注册一些日程，用来使用

//一个日期使用的小案例
#include <map>
void test_time_example()
{
    typedef boost::posix_time::ptime PTime;
    typedef boost::posix_time::time_period time_period;
    PTime pt(boost::gregorian::day_clock::local_day());

    static std::map<time_period, std::string> s_map = {
        {time_period(pt, boost::posix_time::hours(9)),"睡大觉时间"},
        {time_period(pt + boost::posix_time::hours(9), boost::posix_time::hours(3) + boost::posix_time::minutes(30)),"早上上班"},
        {time_period(pt + boost::posix_time::hours(12) + boost::posix_time::minutes(30), boost::posix_time::hours(1)),"午餐时间咯"},
        {time_period(pt + boost::posix_time::hours(13) + boost::posix_time::minutes(30), boost::posix_time::hours(4) + boost::posix_time::minutes(30)),"下午茶时间"},
        {time_period(pt + boost::posix_time::hours(18), boost::posix_time::hours(6)),"争取干完，干不完回家明天继续"}
    };

    auto funcx = [=](PTime pt) {
        for (auto & i : s_map)
        {
            if (i.first.contains(pt))
            {
                std::cout << i.second << std::endl;
            }
        }
    };

    funcx(boost::posix_time::second_clock::local_time());
    funcx(boost::posix_time::second_clock::local_time() + boost::posix_time::hours(16));
}

//C++11自定义字面值特性
boost::gregorian::days operator"" _D(unsigned long long n)
{
    return boost::gregorian::days(n);
}
boost::gregorian::weeks operator"" _W(unsigned long long n)
{
    return boost::gregorian::weeks(n);
}
boost::gregorian::date operator"" _YMD(const char* s, size_t)
{
    return boost::gregorian::from_string(s);
}

}

可能你早都想到了，关于日期格式化，那当然不可能没有啦，这里还能看到C++怎么处理中文输出

//时间日期格式化
void test_time_format()
{
    boost::gregorian::date d(2022, 4, 18);
    boost::gregorian::date_facet* dfact = new boost::gregorian::date_facet("%Y年%m月%d日");
    std::cout.imbue(std::locale(std::cout.getloc(), dfact));
    std::cout << d << std::endl;

    typedef boost::posix_time::ptime PTime;
    boost::posix_time::time_facet* tfact = new boost::posix_time::time_facet("%Y年%m月%d日%H时%M分%S%F秒");
    std::cout.imbue(std::locale(std::cout.getloc(), tfact));
    std::cout << PTime(d, boost::posix_time::hours(12) + boost::posix_time::minutes(30)) << std::endl;
}
