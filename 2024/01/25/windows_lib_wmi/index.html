Windows WMI 技术

一、WMI简介

WMI（Windows Management Instrumentation, Windows管理规范）是Windows 2000/XP管理系统的核心，属于管理数据和操作的基础模块。设计WMI的初衷是达到一种通用性，通过WMI操作系统、应用程序等来管理本地或者远程资源。它支持分布式组件对象模型（DCOM，Distribution Components Object Model）和Windows远程管理（WinRM），用户可通过WMI服务访问、配置、管理和监视Windows所有资源的功能。对于其他的Win32操作系统来讲，WMI是一个非常不错的插件，同时也是测试人员在攻防实战中的一个完美的”无文件攻击“入口途径。

二、WQL

WQL也就是WMI的SQL。WQL的全称是WMI Query Language（Windows管理规范查询语言），主要用于查询WMI的所有托管资源。它的语法与SQL相似，但只能执行数据的查询，不能对类或者实例执行创建、删除、修改等操作。

基础语法：

SELECT properties[, properties] FROM class [where clause]
eg:
SELECT * FROM WIN32_Process where Name Like "%explorer%"

1）SELECT代表WQL语句开始。

2）properties代表要查询的属性名称

3）FROM指定包含SELECT语句中所列出属性的类。

4）class代表要查询的类名称。

5）where clause为可选项，代表要过滤的信息，用来定义搜索范围。

常用查询名：

  • 主机/操作系统信息：Win32_OperatingSystem 、Win32_ComputerSystem
  • 文件/目录列举：CIM_DataFile
  • 磁盘卷列举：Win32_Volume
    • 注册表操作：StdRegProv
  • 运行进程：Win32_Process
  • 服务列举:Win32_Service
  • 事件日志：Win32_NtLogEvent
  • 登录账户：Win32_LoggedOnUser/Win32_LogonSession
  • 共享：Win32_Share
  • 已安装补丁：Win32_QuickFixEngineering
  • 网络信息：Win32_IP4RouteTable
  • 用户账户：Win32_UserAccount
  • 用户组：Win32_Group

三、C++实现WMI查询计算机硬件信息

class SysInfoQuery
{
public:
    bool init(){
        do{
            HRESULT errorCode = -1;
            errorCode = CoInitializeEx(0, COINIT_MULTITHREADED);
            if(FAILED(errorCode)){
                m_needCoUninitialize = false;
                if(RPC_E_CHANGED_MODE != errorCode){
                    //CoInitializeEx error
                    break;
                }
            }

            errorCode = CoInitializeSecurity(NULL, -1, NULL, NULL, RPC_C_AUTHN_LEVEL_PKT_PRIVACY, RPC_C_IMP_LEVEL_IMPERSONATE, NULL, 0, NULL);
            if(FAILED(errorCode)){
                if(RPC_E_TOO_LATE != errorCode){
                    //CoInitializeSecurity error
                    break;
                }
            }
            //创建COM对象
            errorCode = CoCreateInstance(CLSID_WbemAdministrativeLocator, 0, CLSCTX_INPROC_SERVER, IID_IWbemLocator, (LPVOID*)&m_pLoc);
            if(FAILED(errorCode)){
                // CoCreateInstance error
                break;
            }

            errorCode = m_pLoc->ConnectServer(
                _bstr_t(L"ROOT\\CIMV2"), // Object path of WMI namespace
                NULL,                    // User name. NULL = current user
                NULL,                    // User password. NULL = current
                0,                       // Locale. NULL indicates current
                NULL,                    // Security flags.
                0,                       // Authority (e.g. Kerberos)
                0,                       // Context object 
                &m_pSvc);
            if(FAILED(errorCode)){
                //ConnectServer error : %d", errorCode);
                break;
            }

            errorCode = CoSetProxyBlanket(
                m_pSvc,                        // Indicates the proxy to set
                RPC_C_AUTHN_WINNT,           // RPC_C_AUTHN_xxx
                RPC_C_AUTHZ_NONE,            // RPC_C_AUTHZ_xxx
                NULL,                        // Server principal name 
                RPC_C_AUTHN_LEVEL_CALL,      // RPC_C_AUTHN_LEVEL_xxx 
                RPC_C_IMP_LEVEL_IMPERSONATE, // RPC_C_IMP_LEVEL_xxx
                NULL,                        // client identity
                EOAC_NONE                    // proxy capabilities 
                );
            if(FAILED(errorCode)){
            //CoSetProxyBlanket
                break;
            }
        }while(false)

        return TRUE;
    }
    
    HRESULT ExecQuery(const WCHAR* IpSql){
        HRESULT errorCode = -1;
        do{
            if(NULL == m_pSvc){
                break;
            }
            
            errorCode = m_pSvc->ExecQuery(bstr_t(L"WQL"), bstr_t(IpSql), WBEM_FLAG_FORWARD_ONLY | WBEM_FLAG_RETURN_IMMEDIATELY, NULL, &m_pEnumerator);
            if(FAILED(errorCode)){
                //ExecQueryA ExecQuery  error
                break;
            }
            
            uReturn = 0;
            if(m_pEnumerator){
                errorCode = m_pEnumerator->Next(WBEM_INFINITE, 1, &m_pclsObj, &uReturn);
                if(FAILED(errorCode)){
                    //ExecQuery Next error 
                    break;
                }
                if(0 == uReturn){
                    //ExecQuery
                    break;
                }
                return 0;
            }
        }while(false);
        return errorCode;
    }

    HRESULT QueryInfo(const WCHAR* IpName, void* OpBuffer, unsigned int IuBufferLen, bool INeedWcs = false){
        HRESULT errorCode = -1;
        _variant_t vtProp;
        do{


            if(NULL == m_pclsObj || NULL == IpName || NULL == OpBuffer || sizeof(int) > IuBufferLen){
                //QueryInfo m_pclsObj is NULL or param error
                errorCode = ERROR_INVALID_PARAMETER;
                goto Throw;
            }

            errorCode = m_pclsObj->Get(IpName, 0, &vtProp, NULL, 0);
            if(FAILED(errorCode)){
                //QueryInfo Get error 
                goto Throw;
            }

            switch(vtProp.vt){ //value_type
            case VT_NULL:
                *((int*)OpBuffer) = 0;
                break;

            case VT_I4:
                *((int*)OpBuffer) = vtProp.lVal;
                break;

            case wbemCimtypeDatetime: 
            case wbemCimtypeString:
                if(INeedWcs){
                    StringCbCopy((PWCHAR)OpBuffer, IuBufferLen, vtProp.bstrVal);
                }else{
                    if(0 == WideCharToMultiByte(CP_ACP, NULL, vtProp.bstrVal, -1, (char*)OpBuffer, IuBufferLen, NULL, NULL)){
                        errorCode = GetLastError();
                        //QueryInfo WideCharToMultiByte error
                        break;
                    }
                }
                break;

            case wbemCimtypeBoolean:
                *((int*)OpBuffer) = vtProp.boolVal ? 1 : 0;
                break;
            }

            return errorCode;
        }
    }
    
    
private:
    IWbemLocator* m_pLoc;
    IWbemServices* m_pSvc;
    IEnumWbemClassObject* m_pEnumerator;
    IWbemClassObject* m_pclsObj;
};
