Boost 之 UUID

UUID介绍

UUID 是 通用唯一识别码（Universally Unique Identifier）的缩写,UUID是指在一台机器上生成的数字，它保证对在同一时空中的所有机器都是唯一的。通常平台会提供生成的API。按照开放软件基金会(OSF)制定的标准计算，用到了以太网卡地址、纳秒级时间、芯片ID码和随机数。可以轻松不需要中央认证机构创建全球唯一的标识。

UUID可以用在很多地方，数据库记录RawID，标识某个系统的用户，标识网络传输消息等。

基本用法

//uuid == GUID 全面支持比较操作和输入输出流，16个字节
//uui没有统一的头文件、必要的话需要自定义头文件包含他们使用
#include <boost/uuid/uuid.hpp>
#include <boost/uuid/uuid_generators.hpp>
#include <boost/uuid/uuid_io.hpp>
using namespace boost::uuids;
#define msgout(m) std::cout << #m << ":[" << (m) << "]" << std::endl
#include <boost/lexical_cast.hpp>
void test_uuid()
{
    uuid u;
    msgout(uuid::static_size());
    msgout(u.size());

    std::vector<unsigned char> v(16, 7);
    std::copy(v.begin(), v.end(), u.begin());
    msgout(u);

    //全零的uuid认为是无效值
    uuid u1;
    std::fill_n(u1.begin(), u1.size(), 0);
    msgout(u1.is_nil());

    //比较操作被支持
    msgout(u==u1);

    //***********uuid生成器对象*************
    //1.nil_generator
    uuid U1 = nil_generator()();
    msgout(U1);
    msgout(U1==nil_uuid());
    //2.string_generator()格式要匹配，不然抛异常
    msgout(string_generator()("0123456789abcdef0123456789abcdef"));
    msgout(string_generator()("01234567-89ab-cdef-0123-456789abcdef"));
    msgout(string_generator()(L"{01234567-89ab-cdef-0123-456789abcdef}"));
    //3.name_generator() 基于名字的sha1算法，首先提供基准
    uuid U2 = string_generator()("0123456789abcdef0123456789abcdef");
    name_generator ngen(U2);
    uuid U21 = ngen("mario");
    msgout(U21.version() == uuid::version_name_based_sha1);
    msgout(U21);
    uuid U22 = ngen("zdunk");
    msgout(U22);
    //4.随机生成器
    random_generator rgne;
    uuid U3 = rgne();
    msgout(U3.version()==uuid::version_random_number_based);
    msgout(U3);

    //转换，传参时可能会用到
    std::cout << boost::uuids::to_string(U3) << std::endl;
    std::wcout << boost::uuids::to_wstring(U3) << std::endl;
    std::cout << boost::lexical_cast<std::string>(U3) << std::endl;
}

实现一个增强的UUID类

UUID虽然好用，但是生成它却必须使用UUID生成器，可以构封装一个UUID对象，派生boost的uuid类，自动产生对应类型的UUID，简化UUID的使用

#include <boost/uuid/uuid.hpp>
#include <boost/uuid/uuid_generators.hpp>
#include <boost/uuid/uuid_io.hpp>
using namespace boost::uuids;

class uuid_t : public uuid
{
public:
    uuid_t() : uuid(random_generator()()){}		//默认构造生成随机UUID
    uuid_t(int) : uuid(nil_generator()()){}		//构造传int直接生成nil_uuid
    
    uuid_t(const char* str) : uuid(string_generator()(str)){}
    uuid_t(const uuid& u, const char* str) : uuid(name_generator(u)(str)){}
    
    //拷贝构造
    explicit uuid_t(const uuid& u) : uuid(u){}
    
    //类型转换
    operator uuid() {return static_cast<uuid&>(*this);}
    operator uuid() const {return static_cast<const uuid&>(*this);}
    
};

uuid_t的使用方法：

void test_uuid_t()
{
    uuid_t u1;
    uuid_t u2;
    msgout(u1);
    msgout(u2);

    uuid_t u3("0123456789abcdef0123456789abcdef");
    msgout(u3);

    msgout(uuid_t(u3, "zdunk"));
}
